--parar container--
docker stop nome_container
--startar container--
docker start nome_container

--Para fazer dois containers se comunicarem entre si precisa criar uma network para que eles sejam ingressados e se enxerguem nela
vale lembrar que para fazer o container se comunicar na network ja devemos criar ele dentro dela, especificando o parametro --network nome_network
sendo assim é necessário realizar a criação de uma network de maneira antecipada com o comando --docker network create nome_network --

--depois de colocar os conteiners em uma network eles conseguem se comunicar por meio dos seus nomes e portas
